125 template <bool is_mutable>
126 class PetscMeshValues<PetscScalar, 2, is_mutable>
127 {
128  public:
129  using Mesh
130  = typename std::conditional<is_mutable, IPetscMesh<2>, const IPetscMesh<2>>::type;
131  using Reference
132  = typename std::conditional<is_mutable, PetscScalar, const PetscScalar>::type;
133 
134  explicit PetscMeshValues(Mesh& mesh)
135  : m_mesh(&mesh)
136  {
137  using namespace petsc;
138 
139  PetscErrorCode ierr;
140  ierr = DMDAVecGetArray(m_mesh->mesh_spec()->dm(),
141  const_cast<IPetscMesh<2>*>(m_mesh)->petsc_vec(),
142  &m_values);
143  chkerr(ierr);
144 
145  m_ndof = mesh.mesh_spec()->ndof();
146 
147  #ifdef ALLIUM_BOUND_CHECKS
148  m_range = mesh.local_range();
149  #endif
150  }
151  ~PetscMeshValues() {
152  using namespace petsc;
153 
154  if (m_mesh != nullptr) {
155  PetscErrorCode ierr;
156  ierr = DMDAVecRestoreArray(m_mesh->mesh_spec()->dm(),
157  const_cast<IPetscMesh<2>*>(m_mesh)->petsc_vec(),
158  &m_values);
159  chkerr(ierr);
160  }
161  }
162 
163  PetscMeshValues(const PetscMeshValues&) = delete;
164  PetscMeshValues& operator= (const PetscMeshValues&) = delete;
165 
166  PetscMeshValues(PetscMeshValues&& other) {
167  m_values = other.m_values;
168  m_mesh = other.m_mesh;
169  m_ndof = other.m_ndof;
170  #ifdef ALLIUM_BOUND_CHECKS
171  m_range = other.m_range;
172  #endif
173 
174  other.m_mesh = nullptr;
175  }
176 
177  PetscScalar& operator() (int i, int j, int dof = 0) {
178  #ifdef ALLIUM_BOUND_CHECKS
179  if (!m_range.in({i,j}))
180  throw std::logic_error("Out of range access");
181  #endif
182  return m_values[j][i * m_ndof + dof];
183  }
184 
185  int dof_count() const { return m_ndof; }
186  Range<2> local_range() const { return m_mesh->local_range(); }
187 
188  private:
189  typename PetscArrayType<2>::type m_values;
190  Mesh* m_mesh;
191  int m_ndof;
192  #ifdef ALLIUM_BOUND_CHECKS
193  Range<2> m_range;
194  #endif
195 };

197 template <typename N, int D, bool is_mutable>
198 class PetscMeshValues
199 {
200  public:
201  using Mesh = typename PetscMeshValues<PetscScalar, 2, is_mutable>::Mesh;
202 
203  PetscMeshValues(Mesh& mesh)
204  : m_native_values(mesh),
205  m_converted_values(
206  Range<D+1>(mesh.local_range().begin_pos().joined(0),
207  mesh.local_range().end_pos().joined(mesh.mesh_spec()->ndof())))
208  {
209  for (auto p : mesh.local_range()) {
210  for (int i_dof=0; i_dof < m_native_values.dof_count(); ++i_dof) {
211  m_converted_values[p.joined(i_dof)]
212  = narrow_number<N, PetscScalar>()(m_native_values(p[0], p[1], i_dof));
213  }
214  }
215  }
216 
217  PetscMeshValues(PetscMeshValues&& other) = default;
218 
219  ~PetscMeshValues()
220  {
221  for (auto p : m_native_values.local_range()) {
222  for (int i_dof=0; i_dof < m_native_values.dof_count(); ++i_dof) {
223  m_native_values(p[0], p[1], i_dof) = m_converted_values[p.joined(i_dof)];
224  }
225  }
226  }
227 
228  N& operator() (int i, int j, int dof = 0) {
229  return m_converted_values[Point<int, D+1>({i, j, dof})];
230  }
231 
232  private:
233  PetscMeshValues<PetscScalar, 2, is_mutable> m_native_values;
234  LocalMesh<N, D+1> m_converted_values;
235 };
