19  template <typename N>
20  void ImexEulerBase<N>::integrate(Real t1) {
21  allium_assert(t1 >= m_t_cur, "new time value larger than old");
22 
23  auto y_new = allocate_like(*m_y_cur);
24  auto rhs = allocate_like(*m_y_cur);
25 
26  while (m_t_cur < t1) {
27  Real t_new = m_t_cur + m_dt;
28  t_new = std::min(t1, t_new);
29 
30  Real h = t_new - m_t_cur;
31 
32  // rhs = y_cur + h * f_ex(t_cur, y_cur);
33  apply_f_ex(*rhs, m_t_cur, *m_y_cur);
34  *rhs *= h;
35  *rhs += *m_y_cur;
36 
37  // y_cur is a good initial guess to the solver
38  y_new->assign(*m_y_cur);
39 
40  // solve y_new - h * f_im(t_new, y_new) = rhs
41  solve_implicit(*y_new, t_new, h, *rhs, InitialGuess::PROVIDED);
42 
43  m_y_cur.swap(y_new);
44  m_t_cur = t_new;
45  }
46  }
47 
48 }
