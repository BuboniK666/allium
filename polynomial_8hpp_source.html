114  Polynomial derivative() const {
115  std::vector<N> der_coeffs(m_coeffs.size()-1);
116  for (size_t i=0; i < der_coeffs.size(); ++i) {
117  der_coeffs.at(i) = N(i+1) * m_coeffs.at(i+1);
118  }
119 
120  return Polynomial(der_coeffs);
121  }

127  Polynomial anti_derivative() const {
128  std::vector<N> anti_coeffs(m_coeffs.size()+1);
129 
130  anti_coeffs.at(0) = 0;
131  for (size_t i=1; i < anti_coeffs.size(); ++i) {
132  anti_coeffs.at(i) = (N(1)/N(i)) * m_coeffs.at(i-1);
133  }
134 
135  return Polynomial(anti_coeffs);
136  }

172  Polynomial operator* (const Polynomial& other) const {
173  // Initialize coeffs with zero
174  std::vector<N> mul_coeffs(deg() + other.deg() + 1, N(0.0));
175

177  for (size_t j=0; j < other.m_coeffs.size(); ++j) {
178  mul_coeffs.at(i+j) += m_coeffs.at(i) * other.m_coeffs.at(j);
179  }
