48  std::array<PetscInt, 2> ranks_per_dim,

53  {
54  using namespace petsc;
55  PetscErrorCode ierr;
56 
57  ierr = DMDACreate2d(
58  comm.handle(), // comm
59  boundary_type[0], boundary_type[1],
60  stencil_type,
61  global_size[0], global_size[1],
62  ranks_per_dim[0], ranks_per_dim[1],
63  ndof,
64  stencil_width,
65  nullptr, // PetscInt lx[],
66  nullptr, // PetscInt ly[],
67  m_dm.writable_ptr());
68  chkerr(ierr);

70  ierr = DMSetUp(m_dm); chkerr(ierr);

72  DMDALocalInfo info;
73  ierr = DMDAGetLocalInfo(m_dm, &info); chkerr(ierr);
