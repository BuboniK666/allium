73  void set_entries(LocalCooMatrix<Number> mat) override {
74  std::vector<MatrixEntry<N>> entries = std::move(mat).entries();
75 
76  std::vector<MatrixEntry<PetscScalar>> converted(entries.size());
77  std::transform(entries.begin(), entries.end(),
78  converted.begin(),
79  [](MatrixEntry<N> e) -> MatrixEntry<PetscScalar> {
80  return MatrixEntry<PetscScalar>(
81  e.row(),
82  e.col(),
83  e.value());

85 
86  m_native.set_entries(LocalCooMatrix<PetscScalar>(converted));
