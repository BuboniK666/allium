72  template <typename N>
73  LocalMatrix<N>::LocalMatrix(std::initializer_list<std::initializer_list<N>> entries)
74  {
75  const size_t new_rows = entries.size();
76  const size_t new_cols = (new_rows > 0) ? (*entries.begin()).size() : 0;
77 
78  m_storage.resize(new_rows, new_cols);
79 
80  int i = 0;
81  for (auto row : entries) {
82  if (row.size() != new_cols)
83  throw std::runtime_error("Rows are not of equal length.");
84 
85  int j = 0;
86  for (auto entry : row) {
87  m_storage(i, j) = entry;
88  ++j;
89  }
90 
91  ++i;
92  }
93  }
