37 // make a list of objects that can be casted to PetscObject such that we do not
38 // accidentally cast other operators
39 template <typename T>
40 struct is_petsc_object : std::false_type {};
41 
42 template<> struct is_petsc_object<Mat> : std::true_type {};
43 template<> struct is_petsc_object<Vec> : std::true_type {};
44 template<> struct is_petsc_object<KSP> : std::true_type {};
45 template<> struct is_petsc_object<PC> : std::true_type {};
46 template<> struct is_petsc_object<TS> : std::true_type {};
47 template<> struct is_petsc_object<DM> : std::true_type {};

52 template <typename T>
53 std::enable_if_t<is_petsc_object<T>::value, PetscObject>
54 petsc_object_cast(T value) {
55  return reinterpret_cast<PetscObject>(value);
56 }

65 template <typename T>
66 class PetscObjectPtr final {
67  public:
68  static_assert(is_petsc_object<T>::value,
69  "PetscObjectPtr can only hold PETSc objects.");
70 
71  PetscObjectPtr() {
72  value = nullptr;
73  }
74 
75  PetscObjectPtr(const PetscObjectPtr& other) {
76  PetscObjectReference(petsc_object_cast(other.value));
77  this->value = other.value;
78  }
79 
80  PetscObjectPtr(PetscObjectPtr&& other) {
81  this->value = other.value;
82  other.value = nullptr;
83  }
84 
85  PetscObjectPtr(T new_value, bool adopt) {
86  value = nullptr;
87  set(new_value, adopt);
88  }
89 
90  ~PetscObjectPtr() {
91  release();
92  }
93 
94  operator T () const {
95  return get();
96  }
97 
98  PetscObjectPtr& operator= (const PetscObjectPtr& other) {
99  set(other.value, false);
100  return *this;
101  }
102 
105  void release() {
106  if (value != nullptr) {
107  PetscObjectDereference(petsc_object_cast(value));
108  value = nullptr;
109  }
110  }
111 
114  T abandon() {
115  T result = value;
116  value = nullptr;
117  return result;
118  }
119 
120  void set(T new_value, bool adopt) {
121  if (!adopt) {
122  // It is important to increase the reference count first, otherwise we
123  // might free new_value by directly or indirectly releasing it.
124  PetscObjectReference(petsc_object_cast(new_value));
125  }
126  release();
127  value = new_value;
128  }
129 
135  T* writable_ptr() {
136  release();
137  return &value;
138  }
139 
140  T get() const {
141  return value;
142  }
143 
144  friend void swap(PetscObjectPtr<T> &a, PetscObjectPtr<T> &b){
145  T v = a.value;
146  a.value = b.value;
147  b.value = v;
148  }
149 
150  protected:
151  T value;
152 };
